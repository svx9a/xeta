\section{System Architecture}

\subsection{Core Components}

\begin{verbatim}
AETHA/
|-- collector/          # Edge metrics collector
|   |-- agent.ts       # Lightweight Node.js agent
|   `-- worker.ts      # Cloudflare Worker collector
|-- api/               # Sovereign API Layer
|   |-- metrics.ts     # Time-series ingestion
|   |-- alerts.ts      # Alert management
|   `-- auth.ts        # Merchant authentication
|-- dashboard/         # React + TypeScript UI
|   |-- components/    # Reusable sovereign components
|   |-- pages/         # Dashboard views
|   `-- hooks/         # Real-time WebSocket hooks
`-- integrations/      # Data source connectors
    |-- prometheus/    # Prometheus remote write
    |-- cloudflare/    # Cloudflare Analytics
    `-- custom/        # Bring your own metrics
\end{verbatim}

  token: ${AETHA_TOKEN}
  endpoint: https://api.aetha.dev

collectors:
  - type: system
    metrics: [cpu, memory, disk, network]
    interval: 15s

  - type: cloudflare
    zone: example.com
    metrics: [requests, bandwidth, threats]

  - type: custom
    command: ./collect-payment-metrics.sh
    interval: 1m
